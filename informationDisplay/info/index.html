<template>
  <div class="default-main default-wrap register-wrap" v-loading="waitLoading">

            <template #append>

          </el-input>
        </el-form-item>
        <FormItem

          prop="funeralDead.deadGenderCode"

                        keyName: 'sex',

        />
        <FormItem label="年龄" type="number" prop="funeralDead.deadAge" v-model="ruleForm.funeralDead.deadAge"/>
        <FormItem
          label="证件类型"
          type="remoteSelect"

          :input-attr="{

        />
        <FormItem

          type="string"
          :prop="ruleForm.funeralDead.certificateTypeCode === '20' ? '' : 'funeralDead.certificateNo'"

                        onChange: readTheIDNumber,
                    }"

                        disabledDate: disabledDate,
                        onChange: calculateAgeChange,
                    }"
        />
        <FormItem

        <!--      此处数据格式为[1,2,3]  目前后台保存数据需要单个保存，待确定        -->
        <FormItem label="省市区" type="city" prop="funeralDead.provinceCityCountyId"

                        keyName: 'certificates',
                    }"

                  v-model="ruleForm.familyMember.otherPhone"/>
        <FormItem label="联系地址" type="textarea" prop="familyMember.address" v-model="ruleForm.familyMember.address"/>
      </div>

      <div class="default-page default-flex-wrap">
        <el-button type="primary" @click="submitForm(ruleFormRef)" v-auth="'aa'">保存</el-button>
        <el-button type="primary" @click="printBusinessCode">打印逝者二维码</el-button>
      </div>
    </el-form>
  </div>

<script setup lang="ts">
  import {ElMessage, FormInstance} from 'element-plus'

  import {onMounted, reactive, ref, watch} from 'vue'
  import {UseServiceInfo} from '@/stores/service'
  import {
    loadPrintCremationArchivesCodeWithBusinessCode,
    loadWithBusinessCode,
    saveCremationArchives
  } from '@/api/service'
  import {buildValidatorData} from '@/utils/validate'
  import {calculateAge, extractInfoFromIDCard, getNowDate} from '@/utils'
  import {printDeadCode} from '@/utils/printTheDocument'
  import {useIdCardWebSocket} from '@/hooks/useIdCardWebSocket'

  const {idCardInfo, readIdCard} = useIdCardWebSocket()

  const businessCode = ref('')
  const waitLoading = ref(false)

  const ruleFormRef = ref<FormInstance>()
  const serviceInfo = UseServiceInfo()
  if (serviceInfo.getBusinessCode() !== '') {
    businessCode.value = serviceInfo.getBusinessCode()
  }
  const rules = reactive({
    'funeralDead.deadName': [
      buildValidatorData({
        name: 'required',
        message: '请输入逝者姓名',
      }),
    ],
    'funeralDead.certificateNo': [
      buildValidatorData({
        name: 'required',
        message: '请输入证件号码',
      }),
      buildValidatorData({
        name: 'certificateNo',
      }),
    ],
    'familyMember.mobilePhone': [
      buildValidatorData({
        name: 'required',
        message: '请输入手机号',
      }),
      buildValidatorData({
        name: 'mobile',
      }),
    ],
  })
  let ruleForm = reactive({
    funeralDead: {
      businessCode: '',
      deadName: '',
      deadGenderCode: '',
      deadAge: '',
      certificateTypeCode: '1',
      certificateNo: '',
      birthDate: '',
      deathDate: '',
      provinceCityCountyId: '',
      beforeDeathAddress: '',
      remainsOriginCode: '',
      deathCause: '',
      asArrived: '1',
      arrivedTime: getNowDate('time'),
      createUserName: '',
      remark: '',
    },
    familyMember: {
      memberName: '',
      certificateTypeCode: '1',
      certificateNo: '',
      memberRelation: '',
      mobilePhone: '',
      otherPhone: '',
      address: '',
    },
  })
  //禁止选择今天以后的时间

  const disabledDate = (time) => {
    return time.getTime() > Date.now()
  }
  // 查询业务信息
  const onSearchInfo = async () => {
    if (!businessCode.value) {
      ElMessage.warning('请输入业务编码')
      return
    }
    serviceInfo.setBusinessCode(businessCode.value)
    waitLoading.value = true
    try {
      let res = await loadWithBusinessCode({businessCode: businessCode.value})
      ruleForm.funeralDead = res.data.funeralDead
      ruleForm.familyMember = res.data.familyMember
      waitLoading.value = false
    } catch (e) {
      waitLoading.value = false
    }
  }
  // 表单提交
  const submitForm = async (formEl: FormInstance | undefined) => {
    if (!formEl) return
    await formEl.validate((valid, fields) => {
      if (valid) {
        //判断家属和承办人证号
        if (ruleForm.funeralDead.certificateNo && ruleForm.familyMember.certificateNo && ruleForm.funeralDead.certificateNo === ruleForm.familyMember.certificateNo) {
          ElMessage('逝者、家属或承办人证件号码不能一致')
          return false
        }
        saveFormData()
      } else {
        console.log('error submit!', fields)
      }
    })
  }
  // 业务信息保存
  const saveFormData = async () => {
    waitLoading.value = true
    let params = ruleForm
    try {
      let res = await saveCremationArchives(params)
      console.log(res.code)
      ElMessage({
        message: res.message,
        type: 'success',
      })
      waitLoading.value = false
      onSearchInfo()
    } catch (e) {
      waitLoading.value = false
    }
  }
  // 表单重置
  const resetForm = (formEl: FormInstance | undefined) => {
    if (!ruleFormRef.value) return
    ruleFormRef.value.resetFields()
  }
  // 单条直接input调用
  const readTheIDNumber = (value, form = 'funeralDead') => {
    let res = extractInfoFromIDCard(value)
    if (res) {
      let {birthday, age} = res
      ruleForm[form].birthDate = birthday
      ruleForm[form].deadAge = age
    }
  }

  // 多条数据的  监听身份证，返回出生日期
  // watch(
  //     () => {
  //         return [ruleForm.funeralDead.certificateNo, ruleForm.familyMember.certificateNo]
  //     },
  //     ([funeralDeadCertNo, familyMemberCertNo]) => {
  //         console.log(funeralDeadCertNo)
  //         readTheIDNumber(funeralDeadCertNo, 'funeralDead')
  //         console.log(familyMemberCertNo)
  //         readTheIDNumber(familyMemberCertNo, 'familyMember')
  //     }
  // )

  // 打印逝者二维码
  const printBusinessCode = async () => {
    let res = await loadPrintCremationArchivesCodeWithBusinessCode({businessCode: businessCode.value})
    printDeadCode(res.data.funeralDead)
  }
  // 读取身份证
  const readCardType = ref('')
  const handleReadIdCard = (type) => {
    try {
      readCardType.value = type
      readIdCard()
    } catch (e) {
      console.log(e)
    }
  }
  //读取身份证信息后赋值
  const processingCardInfo = (IdCardInfo) => {
    const {name, deadGender, nation, birthDate, householdRegistrationPlace, idCardNo} = IdCardInfo

    if (readCardType.value === 'funeralDead') {
      ruleForm.funeralDead.deadName = name
      ruleForm.funeralDead.birthDate = birthDate
      ruleForm.funeralDead.beforeDeathAddress = householdRegistrationPlace
    } else if (readCardType.value === 'familyMember') {
      ruleForm.familyMember.memberName = name
      ruleForm.familyMember.certificateNo = birthDate
      ruleForm.familyMember.address = householdRegistrationPlace
    }
  }
  // 监视 idCardInfo 对象的变化
  watch(
    () => idCardInfo,
    (newVal) => {
      console.log('身份证信息已更新', newVal)
      processingCardInfo(newVal)
    },
    {deep: true}
  )
  // 计算年龄
  const calculateAgeChange = () => {
    let birthDate = ruleForm.funeralDead.birthDate
    let deathDate = ruleForm.funeralDead.deathDate
    console.log(birthDate, deathDate)
    ruleForm.funeralDead.deadAge = calculateAge(birthDate, deathDate)
  }

  onMounted(() => {
    console.log()
  })
</script>
